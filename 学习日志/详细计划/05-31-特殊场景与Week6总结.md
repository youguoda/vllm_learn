画多轮 TTFT 折线图 `assets/multi_turn_ttft.png`：

```python
# plot_multi_turn.py
import csv, matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, axes = plt.subplots(1, 3, figsize=(15, 5))
    concs = [1, 8, 16]
    for ci, conc in enumerate(concs):
        ax = axes[ci]
        ax.set_title(f"并发={conc}")
        for fw, c, s in (("vllm","#4C72B0","o-"), ("sglang","#55A868","s--")):
            try:
                with open(f"bench_results/{fw}_multi_turn.csv") as f:
                    rows = list(csv.DictReader(f))
                data = [r for r in rows
                        if int(r["concurrency"])==conc
                        and r["turn"] != "throughput"]
                x = [int(r["turn"]) for r in data]
                y = [float(r["ttft_median"]) for r in data]
                ax.plot(x, y, s, color=c, label=fw, ms=7)
            except FileNotFoundError:
                pass
        ax.set_xlabel("对话轮次")
        ax.set_ylabel("TTFT (ms)" if ci==0 else "")
        ax.legend()
        ax.axhline(0, color='gray', ls=':', lw=0.5)
    fig.suptitle("多轮对话 TTFT 变化（SGLang RadixAttention 主场）", fontsize=13)
    fig.tight_layout()
    fig.savefig("assets/multi_turn_ttft.png", dpi=150, bbox_inches="tight")
```

**预期**：vLLM 每轮 TTFT 随历史增长而上升（每轮要 prefill 越来越长的历史）；SGLang 第 2 轮起 TTFT 降低并保持稳定（历史被 radix 树缓存，只 prefill 新问题）。差异在高并发下更明显。

# test_long_ctx.py
from vllm import LLM, SamplingParams
import time

if __name__ == "__main__":
    for max_len in (2048, 4096):
        llm = LLM("Qwen/Qwen2.5-1.5B", max_model_len=max_len,
                  gpu_memory_utilization=0.85)
        # 构造接近 max_len 的输入
        long_input = "请分析以下传感器数据序列：" + "振动值=100, " * (max_len // 5)
        long_input = long_input[:max_len * 3 - 200]  # char≈token×3
        params = SamplingParams(max_tokens=128, temperature=0.0)
        t0 = time.time()
        out = llm.generate([long_input], params)
        print(f"max_len={max_len}: 耗时={time.time()-t0:.1f}s, "
              f"输出={len(out[0].outputs[0].token_ids)} token")
        del llm
        import torch; torch.cuda.empty_cache()
