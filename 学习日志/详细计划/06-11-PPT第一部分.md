```python
# fill_part1.py — 填充 Part 1 的四张幻灯片（在骨架基础上修改）
from pptx import Presentation
from pptx.util import Inches, Pt, Emu
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN
from pptx.util import Inches
import matplotlib.pyplot as plt
import numpy as np, io, os

# ── 工具函数 ─────────────────────────────────────────────────────

def add_title(slide, text, y=0.4, size=32):
    txb = slide.shapes.add_textbox(Inches(0.8), Inches(y), Inches(11.5), Inches(1.0))
    tf = txb.text_frame
    p = tf.paragraphs[0]
    p.text = text
    p.alignment = PP_ALIGN.LEFT
    p.runs[0].font.size = Pt(size)
    p.runs[0].font.bold = True
    p.runs[0].font.color.rgb = RGBColor(0x1F,0x4E,0x79)

def add_body(slide, lines, x=0.8, y=1.6, w=11.5, h=4.5, size=20):
    txb = slide.shapes.add_textbox(Inches(x), Inches(y), Inches(w), Inches(h))
    tf = txb.text_frame
    tf.word_wrap = True
    for i, line in enumerate(lines):
        p = tf.paragraphs[0] if i == 0 else tf.add_paragraph()
        p.text = line
        p.runs[0].font.size = Pt(size)
        p.space_after = Pt(6)

def add_image(slide, img_path, x, y, w, h):
    if os.path.exists(img_path):
        slide.shapes.add_picture(img_path, Inches(x), Inches(y), Inches(w), Inches(h))
    else:
        # 占位符方框
        txb = slide.shapes.add_textbox(Inches(x), Inches(y), Inches(w), Inches(h))
        txb.text_frame.text = f"[图：{img_path}]"

def make_gpu_util_bar(util_pct=35):
    """生成 GPU 利用率示意图（内联，不依赖外部图片）"""
    fig, ax = plt.subplots(figsize=(6, 2))
    ax.barh(["prefill", "decode"], [95, util_pct], color=["#2196F3","#FF5722"])
    ax.set_xlim(0, 100)
    ax.set_xlabel("GPU 利用率 (%)")
    ax.axvline(100, color="gray", linestyle="--", alpha=0.5, label="理论峰值")
    ax.set_title("朴素推理：decode 阶段 GPU 严重闲置", fontsize=11)
    for bar, val in zip(ax.patches, [95, util_pct]):
        ax.text(val + 1, bar.get_y() + bar.get_height()/2,
                f"{val}%", va="center", fontsize=11)
    plt.tight_layout()
    buf = io.BytesIO(); fig.savefig(buf, format="png", dpi=150); plt.close(); buf.seek(0)
    return buf

def make_kv_waste_diagram():
    """KV Cache 内存浪费图"""
    fig, axes = plt.subplots(1, 2, figsize=(8, 3))
    # 左：传统预分配
    seqs = [0.3, 0.6, 0.15, 0.8, 0.5]  # 实际使用比例
    colors_use  = ["#4CAF50"] * 5
    colors_waste = ["#FF5722"] * 5
    for i, (use, waste) in enumerate(zip(seqs, [1-s for s in seqs])):
        axes[0].bar(i, use,   color="#4CAF50", label="已用" if i==0 else "")
        axes[0].bar(i, waste, bottom=use, color="#FF5722", label="浪费" if i==0 else "")
    axes[0].set_ylim(0, 1.2)
    axes[0].set_title("传统预分配：平均浪费 ~60%")
    axes[0].set_xticks(range(5)); axes[0].set_xticklabels([f"Seq{i}" for i in range(5)])
    axes[0].legend(loc="upper right", fontsize=8)
    axes[0].set_ylabel("内存占比")
    # 右：PagedAttention
    axes[1].bar(range(5), seqs, color="#4CAF50")
    axes[1].bar(range(5), [0.05]*5, bottom=seqs, color="#FF5722")
    axes[1].set_ylim(0, 1.2)
    axes[1].set_title("PagedAttention：浪费 <4%（尾块）")
    axes[1].set_xticks(range(5)); axes[1].set_xticklabels([f"Seq{i}" for i in range(5)])
    axes[1].set_ylabel("内存占比")
    plt.tight_layout()
    buf = io.BytesIO(); fig.savefig(buf, format="png", dpi=150); plt.close(); buf.seek(0)
    return buf

# ── 主程序 ────────────────────────────────────────────────────────

if __name__ == "__main__":
    prs = Presentation("学习日志/PPT骨架.pptx")
    # Part 1 对应幻灯片索引 2,3,4（0-indexed），第2张是"GPU 利用率"
    # 注意：骨架里幻灯片顺序从 0 开始

    # 幻灯片 2：GPU 利用率问题
    slide = prs.slides[2]
    slide.shapes._spTree.clear()  # 清空骨架内容，重新填
    add_title(slide, "Step 0：朴素推理为什么低效？")
    add_body(slide, [
        "• decode 是自回归的：每次只生成一个 token，GPU 利用率极低",
        "• 实测（Qwen2.5-1.5B，单请求）：decode 阶段 GPU 利用率仅约 35%",
        "• 根本原因：decode 是 memory-bandwidth bound，不是 compute bound",
        "",
        "→ 解决思路：Continuous Batching（让 GPU 同时处理多条请求的 decode）",
    ])
    img_buf = make_gpu_util_bar(util_pct=35)
    slide.shapes.add_picture(img_buf, Inches(3), Inches(3.8), Inches(7), Inches(2.5))

    # 幻灯片 3：KV Cache 是核心瓶颈
    slide = prs.slides[3]
    slide.shapes._spTree.clear()
    add_title(slide, "KV Cache：为什么是推理的核心瓶颈？")
    add_body(slide, [
        "• 每个 token 的 attention 需要看所有历史 token 的 K、V 向量",
        "• 为避免重算，把 K/V 存下来 → KV Cache",
        "• 内存占用随序列长度线性增长：",
        "    1 token = 2 × num_layers × num_heads × head_dim × 2 bytes",
        "    Qwen2.5-1.5B，max_len=4096：KV Cache 约 6 GB",
        "• 传统方案：按最大长度预分配 → 大量浪费",
    ])

    # 幻灯片 4：KV Cache 浪费分析（对比图）
    slide = prs.slides[4]
    slide.shapes._spTree.clear()
    add_title(slide, "传统 KV Cache：平均浪费 60-80%")
    img_buf = make_kv_waste_diagram()
    slide.shapes.add_picture(img_buf, Inches(1.5), Inches(1.5), Inches(10), Inches(4))
    add_body(slide, [
        "• 内部碎片：序列不足最大长度，尾部空闲",
        "• 外部碎片：整块预分配，中间无法复用",
        "• vLLM 的 PagedAttention：把 KV Cache 分成 16-token 的 block，按需分配",
    ], y=5.8, size=16)

    prs.save("学习日志/PPT_Part1.pptx")
    print("Part 1 填充完成 → PPT_Part1.pptx")
```

```python
from pptx import Presentation
from pptx.util import Inches, Pt
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN
import json, os

def add_title(slide, text, size=32):
    txb = slide.shapes.add_textbox(Inches(0.8), Inches(0.4), Inches(11.5), Inches(1.0))
    tf = txb.text_frame

def add_body(slide, lines, x=0.8, y=1.6, w=11.5, h=4.0, size=20):
    txb = slide.shapes.add_textbox(Inches(x), Inches(y), Inches(w), Inches(h))
    tf = txb.text_frame; tf.word_wrap = True
    for i, line in enumerate(lines):
        p = tf.paragraphs[0] if i == 0 else tf.add_paragraph()
        p.text = line; p.runs[0].font.size = Pt(size); p.space_after = Pt(5)

def add_image(slide, img_path, x, y, w, h):
    if os.path.exists(img_path):
        slide.shapes.add_picture(img_path, Inches(x), Inches(y), Inches(w), Inches(h))
    else:
        txb = slide.shapes.add_textbox(Inches(x), Inches(y), Inches(w), Inches(h))
        txb.text_frame.text = f"[图片占位: {os.path.basename(img_path)}]"

if __name__ == "__main__":
    # 从 all_data.json 读取 APC 实验数据（如果有）
    apc_cold = apc_warm = "___"
    try:
        with open("bench_results/all_data.json") as f:
            D = json.load(f)
        # 尝试读取前缀缓存实验（06_prefix_caching.py 的结果）
        if "prefix_cache" in D:
            apc_cold = f"{D['prefix_cache']['cold_ttft']:.0f}ms"
            apc_warm = f"{D['prefix_cache']['warm_ttft']:.0f}ms"
    except: pass

    prs = Presentation("学习日志/PPT_Part1.pptx")

    # 幻灯片 5：PagedAttention
    slide = prs.slides[5]
    slide.shapes._spTree.clear()
    add_title(slide, "PagedAttention：向 OS 借来的内存管理思想")
    add_body(slide, [
        "• 类比操作系统的虚拟内存：把物理内存分成固定大小的 page",
        "• KV Cache 分成 block（每 block = 16 token 的 K/V）",
        "• Block Table：逻辑 block → 物理 block 的映射",
        "• 内存浪费从 60-80% 降到 <4%（仅尾 block 不满）",
        "• 支持 Copy-on-Write：多条请求共享同一个 prefix block",
    ], size=19)
    add_image(slide, "assets/pagedattention_arch.png", 7.5, 1.5, 5.5, 5.0)

    # 幻灯片 6：Continuous Batching
    slide = prs.slides[6]
    slide.shapes._spTree.clear()
    add_title(slide, "Continuous Batching：动态插入，消灭气泡")
    add_body(slide, [
        "• 传统 static batching：一批请求全部完成才接受下一批",
        "• Continuous batching：每个 iteration 检查是否有空位，立刻插入新请求",
        "• 效果：GPU 持续满负载，长请求不会阻塞短请求",
        "• vLLM 调度器：WAITING → RUNNING → PREEMPTED 三状态",
        "    抢占策略：recompute（重算）或 swap（换出到 CPU）",
    ], size=19)
    add_image(slide, "assets/scheduler_compare.png", 7.5, 1.5, 5.5, 4.5)

    add_body(slide, [

        "• 相同前缀（system prompt、few-shot）→ 哈希命中 → 跳过 prefill",
        f"• 实测（1000-token 前缀，Qwen2.5-1.5B）：",
        f"    冷启动 TTFT：{apc_cold}  →  缓存命中 TTFT：{apc_warm}",
        "• 关键：前缀必须对齐到 16-token block 边界",
        "    不满一 block 的尾部不能复用（SGLang 的改进点）",

        "前缀复用        APC               system prompt TTFT 降幅 ___\%",
        "",
        "三者协同 → 同等硬件下，相比朴素推理：吞吐 ___x，单请求延迟 ___ms",
```

```markdown
# 演讲稿草稿——Part 1（约 3 分钟）

## 幻灯片2：GPU 利用率
开场白：（展示 GPU 利用率图）
"这张图是我用 nvidia-smi 实际测的。同一张 RTX 3060 的卡，prefill 时
GPU 跑到 95%，但一切换到 decode，掉到 35%。为什么？因为 decode 每次只生成
一个 token，大量 CUDA core 在等待内存读取，完全是内存带宽瓶颈。"

过渡："那内存里在存什么？这就引出 KV Cache。"

## 幻灯片3：KV Cache
"每次 decode 时，注意力机制要看所有历史 token 的 Key 和 Value 向量——
这些向量就是 KV Cache。以 Qwen2.5-1.5B 为例，最大长度 4096 token，
KV Cache 就要占 6GB——对于只有 12GB 显存的卡，这是大头。"

## 幻灯片4：浪费分析
"传统做法是按最大长度预分配——序列不够长，后面就空着。
平均浪费 60-80%。有时候一张卡上只能跑两三条请求，不是算力不够，
是内存全被'占位'占完了。这就是 vLLM 的切入点。"

过渡："vLLM 怎么解决的？三个层次——PagedAttention、Continuous Batching、APC。"
```
