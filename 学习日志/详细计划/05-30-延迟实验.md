```python
# plot_latency.py
import csv, matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, ax = plt.subplots(figsize=(9, 5))
    for fw, c, s in (("vllm","#4C72B0","o-"), ("sglang","#55A868","s--")):
        try:
            with open(f"bench_results/{fw}_latency.csv") as f:
                rows = list(csv.DictReader(f))
            x  = [int(r["input_len_approx"]) for r in rows]
            y  = [float(r["ttft_median"])     for r in rows]
            yp = [float(r["ttft_p95"])        for r in rows]
            ax.plot(x, y, s, color=c, label=f"{fw} 中位数", ms=7)
            ax.fill_between(x, y, yp, alpha=0.15, color=c, label=f"{fw} P95")
        except FileNotFoundError:
            print(f"{fw} 数据不存在")
    ax.set_xlabel("输入长度（近似 token 数）")
    ax.set_ylabel("TTFT (ms)")
    ax.set_title("TTFT vs 输入长度（低并发，prefill 主导）")
    ax.legend()
    # 添加理论线：TTFT ∝ input_len（prefill 线性）
    if rows:
        import numpy as np
        x_arr = [int(r["input_len_approx"]) for r in rows]
        y_arr = [float(r["ttft_median"])     for r in rows]
        slope = y_arr[-1] / x_arr[-1]
        ax.plot(x_arr, [slope*xi for xi in x_arr], 'k:', lw=1,
                label="理论线性（参考）")
    ax.legend()
    fig.tight_layout()
    fig.savefig("assets/latency_ttft.png", dpi=150, bbox_inches="tight")
```

**重点观察**：TTFT 是否与输入长度近似线性（说明 prefill 是主导）？两框架斜率是否一致（说明 prefill 算法无本质区别）？

```python
# plot_itl.py
import csv, matplotlib.pyplot as plt
import statistics
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, axes = plt.subplots(1, 3, figsize=(13, 5))
    concs = [1, 4, 8]
    for ci, conc in enumerate(concs):
        ax = axes[ci]
        ax.set_title(f"并发={conc}")
        data = {}
        for fw in ("vllm", "sglang"):
            try:
                with open(f"bench_results/{fw}_streaming.csv") as f:
                    rows = list(csv.DictReader(f))
                row = next(r for r in rows if int(r["concurrency"]) == conc)
                data[fw] = [float(row["itl_p50"]), float(row["itl_p95"]),
                            float(row["itl_p99"])]
            except (FileNotFoundError, StopIteration):
                data[fw] = [0, 0, 0]
        x = [0.8, 1.2]
        for xi, fw in zip(x, ("vllm","sglang")):
            d = data[fw]
            ax.bar(xi, d[2], width=0.3, color=("#f8d7da" if fw=="vllm" else "#d4edda"),
                   label=f"{fw} P99", edgecolor='gray')
            ax.bar(xi, d[1], width=0.3, color=("#4C72B0" if fw=="vllm" else "#55A868"),
                   alpha=0.7, label=f"{fw} P95")
            ax.bar(xi, d[0], width=0.3, color=("#1a3a6b" if fw=="vllm" else "#1a5c34"),
                   label=f"{fw} P50")
            ax.text(xi, d[0]+2, f"P50\n{d[0]:.0f}", ha='center', fontsize=7)
        ax.set_xticks(x); ax.set_xticklabels(["vllm","sglang"])
        ax.set_ylabel("ITL (ms/token)" if ci==0 else "")
    fig.suptitle("ITL 分布（token 间延迟，越低打字越流畅）", fontsize=13)
    fig.tight_layout()
    fig.savefig("assets/itl_distribution.png", dpi=150, bbox_inches="tight")
```

**关键指标 ITL（Inter-Token Latency）**：用户感受到的"打字速度"。P50 决定通常体验，P99 决定最差体验（卡顿）。
