import re

def parse_gpu_log(path):
    """解析 nvidia-smi csv 输出"""
    rows = []
    with open(path) as f:
        reader = csv.reader(f)
        next(reader)  # 跳过 header
        for row in reader:
            try:
                gpu_util = float(row[1].strip().replace(' %',''))
                mem_used  = float(row[3].strip().replace(' MiB',''))
                rows.append({"gpu_util": gpu_util, "mem_used": mem_used})
            except (ValueError, IndexError):
                pass
    return rows

fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(13, 5))
for fw, color in (("vllm","#4C72B0"),("sglang","#55A868")):
    p = f"bench_results/gpu_{fw}.csv"
    try:
        rows = parse_gpu_log(p)
        t = list(range(len(rows)))
        ax1.plot(t, [r["gpu_util"] for r in rows],

        ax2.plot(t, [r["mem_used"] for r in rows],

## 步骤 3：找最优负载区间（30 min）

```python
# find_optimal_range.py — 找吞吐/延迟性价比最高的并发数
import json

with open("bench_results/all_data.json") as f:
    D = json.load(f)

if __name__ == "__main__":
    print("=== 最优负载区间分析 ===\n")
    for fw in ("vllm","sglang"):
        rows = D.get("fixed_output",{}).get(fw,[])
        if not rows: continue
        print(f"--- {fw} ---")
        # 计算"效率"= tok/s / (TTFT_ms / 1000)，越大越好
        for r in rows:
            tps  = r.get("tok_per_s", 0)
            ttft = r.get("ttft_median", 1) / 1000  # 转秒
            eff  = tps / ttft if ttft > 0 else 0
            print(f"  并发={int(r['concurrency'])}: "
                  f"吞吐={tps:.1f} tok/s, TTFT={r.get('ttft_median',0):.0f}ms, "
                  f"效率={eff:.1f}")
        # 找效率最高的并发
        best = max(rows, key=lambda r: r.get("tok_per_s",0) /
                   max(r.get("ttft_median",1)/1000, 0.01))
        print(f"  → 最优并发: {int(best['concurrency'])}")
```
