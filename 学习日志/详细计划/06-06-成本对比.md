## 步骤 1：用实验数据算"卡时成本"（40 min）

```python
# cost_analysis.py
import json

with open("bench_results/all_data.json") as f:
    D = json.load(f)

# 云 GPU 参考价格（查当前价，填入）
# 示例：RTX 3090 级别 ~¥3/h，A10 ~¥5/h（按需价，实际查询）
GPU_PRICE_PER_HOUR_RMB = 3.0  # 填入你查到的当前价格

if __name__ == "__main__":
    print("=== 成本分析：每百万 token 成本估算 ===\n")
    print(f"GPU 价格假设: ¥{GPU_PRICE_PER_HOUR_RMB}/小时\n")

    results = {}
    for fw in ("vllm","sglang"):
        rows = D.get("fixed_output",{}).get(fw,[])
        if not rows: continue
        best = max(rows, key=lambda r: r.get("tok_per_s",0))
        tps = best["tok_per_s"]
        # 每小时产出 token 数
        tokens_per_hour = tps * 3600
        # 每百万 token 成本（元）
        cost_per_million = GPU_PRICE_PER_HOUR_RMB / (tokens_per_hour / 1_000_000)
        results[fw] = {"tps": tps, "conc": best["concurrency"],
                       "tokens_per_hour": tokens_per_hour,
                       "cost_per_million": cost_per_million}
        print(f"{fw}:")
        print(f"  最优并发: {int(best['concurrency'])}")
        print(f"  最高吞吐: {tps:.1f} tok/s")
        print(f"  每小时产出: {tokens_per_hour/1e6:.2f} M token")
        print(f"  每百万 token 成本: ¥{cost_per_million:.3f}")

    if len(results) == 2:
        diff_pct = (results["vllm"]["cost_per_million"] -
                    results["sglang"]["cost_per_million"]) / \
                    results["vllm"]["cost_per_million"] * 100
        print(f"\n无共享前缀场景：SGLang 成本{'低' if diff_pct>0 else '高'} {abs(diff_pct):.1f}%（通用负载下差异小）")

    print("\n=== 多轮对话场景成本估算 ===")
    print("（多轮对话中 SGLang 减少了 prefill，等效提升吞吐，成本随之降低）")
    # 简单估算：假设多轮对话 SGLang 的平均 TTFT 是 vLLM 的 X%
    # 则等效吞吐提升约 1/X 倍，成本降低约 (1-X) %
    for fw in ("vllm","sglang"):
        rows = [r for r in D.get("multi_turn",{}).get(fw,[])
                if r.get("concurrency")==8 and str(r.get("turn","")).isdigit()]
        if rows:
            avg_ttft = sum(r["ttft_median"] for r in rows) / len(rows)
            print(f"  {fw} 多轮平均 TTFT: {avg_ttft:.0f}ms")
```

- 通用负载：两框架成本差异 ___\%，与吞吐差异等比例
