```yaml
# bench_config.yaml — Week 6 全部实验的控制变量文档

model: "Qwen/Qwen2.5-1.5B"
max_model_len: 4096
gpu_memory_utilization: 0.85   # 两框架对应参数需对齐

# 实验场景矩阵
scenarios:
  # 场景A：固定输出（测吞吐上限）
  fixed_output:
    input_len: 256         # 固定输入长度
    output_len: 128        # 固定输出长度
    concurrency: [1, 4, 8, 16, 32]
    n_requests: 40         # 每组请求数

  # 场景B：延迟测试（TTFT 和 ITL）
  latency:
    input_lens: [64, 256, 512, 1024, 2048]
    output_len: 64         # 固定输出，主要测 prefill 耗时(TTFT)
    concurrency: 1         # 低并发，隔离延迟
    n_requests: 10

  # 场景C：多轮对话（RadixAttention 主场）
  multi_turn:
    n_turns: 5             # 每个对话轮数
    system_prompt_tokens: 500   # 共享前缀长度
    user_tokens_per_turn: 50
    concurrency: [1, 8, 16]
    n_dialogs: 20          # 对话数量

  # 场景D：流式输出 ITL（打字体验）
  streaming:
    input_len: 256
    output_len: 256
    concurrency: [1, 4, 8]
    n_requests: 20

# 重复次数（取中位数）
repeats: 3

# 预热（不计入结果）
warmup_requests: 5

# 单卡注意事项
notes:
  - "并发 >32 在 12GB 显存下会严重排队，上限 32"
  - "每轮实验间重启 server，清 EngineCore 残留进程"
  - "两框架启动参数必须对齐 max_model_len / gpu_memory_utilization"
```

## 步骤 3：编写自动化测试框架基础（30 min）

新建 `12_bench_runner.py`（今天只写骨架，明天补完）：

```python
"""
Week 6 统一压测框架
用法:
  NO_PROXY="*" python3 12_bench_runner.py --framework vllm   --scenario fixed_output
  NO_PROXY="*" python3 12_bench_runner.py --framework sglang --scenario fixed_output
"""
import asyncio, csv, time, argparse, statistics, json
from pathlib import Path
import openai

# ── 配置 ──────────────────────────────────────────────────────
BASE_URLS = {"vllm": "http://localhost:8000/v1",
             "sglang": "http://localhost:30000/v1"}
MODEL = "Qwen/Qwen2.5-1.5B"
OUTPUT_DIR = Path("bench_results")
OUTPUT_DIR.mkdir(exist_ok=True)

# ── 单请求计时 ────────────────────────────────────────────────
async def timed_request(client, messages, max_tokens, stream=False):
    t0 = time.perf_counter()
    ttft = None; tokens = 0; chunks = []
    r = await client.chat.completions.create(
        model=MODEL, messages=messages,
        max_tokens=max_tokens, stream=stream)
    if stream:
        async for c in r:
            if ttft is None: ttft = time.perf_counter() - t0
            d = c.choices[0].delta.content
            if d: tokens += 1; chunks.append(d)
    else:
        ttft = time.perf_counter() - t0
        tokens = len(r.choices[0].message.content) // 4  # 粗估
    e2e = time.perf_counter() - t0
    itl = (e2e - ttft) / max(tokens - 1, 1)
    return {"ttft": ttft, "e2e": e2e, "itl": itl,
            "tokens": tokens, "text": "".join(chunks) if stream else ""}

# ── 场景 A：固定输出吞吐 ──────────────────────────────────────
def make_fixed_prompt(input_len=256):
    base = "请分析以下工业传感器数据并给出维护建议。数据如下：振动频率{f}Hz，温度{t}℃，噪声水平{n}dB。"
    return base.format(f=300, t=65, n=82) + "请详细说明。" * (input_len // 15)

async def bench_fixed_output(client, concurrency, n_requests, output_len):
    sem = asyncio.Semaphore(concurrency)
    msgs = [{"role": "user", "content": make_fixed_prompt()}]
    async def one():
        async with sem:
            return await timed_request(client, msgs, output_len, stream=True)
    t0 = time.perf_counter()
    results = await asyncio.gather(*[one() for _ in range(n_requests)])
    wall = time.perf_counter() - t0
    return {
        "concurrency": concurrency,
        "rps": n_requests / wall,
        "tok_per_s": sum(r["tokens"] for r in results) / wall,
        "ttft_median": statistics.median(r["ttft"] for r in results) * 1000,
        "ttft_p95":    sorted(r["ttft"] for r in results)[int(n_requests*0.95)] * 1000,
        "itl_median":  statistics.median(r["itl"] for r in results) * 1000,
        "e2e_median":  statistics.median(r["e2e"] for r in results) * 1000,
    }

# ── 主入口（今天只跑固定输出场景作验证）────────────────────────
if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--framework", choices=["vllm","sglang"], required=True)
    ap.add_argument("--scenario",  default="fixed_output")
    a = ap.parse_args()

    client = openai.AsyncOpenAI(base_url=BASE_URLS[a.framework], api_key="x")

    # 预热
    print("预热中…")
    asyncio.run(bench_fixed_output(client, concurrency=2, n_requests=5, output_len=64))

    if a.scenario == "fixed_output":
        rows = []
        for conc in (1, 4, 8, 16, 32):
            print(f"  并发={conc}…", end="", flush=True)
            r = asyncio.run(bench_fixed_output(client, conc, 40, 128))
            r["framework"] = a.framework
            print(f" RPS={r['rps']:.2f}, TTFT={r['ttft_median']:.0f}ms")
            rows.append(r)

        out = OUTPUT_DIR / f"{a.framework}_fixed_output.csv"
        with open(out, "w", newline="") as f:
            w = csv.DictWriter(f, fieldnames=rows[0].keys())
            w.writeheader(); w.writerows(rows)
        print(f"\n保存到 {out}")
```
