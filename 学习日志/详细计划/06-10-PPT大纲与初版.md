## 步骤 2：制作骨架 PPT（120 min）

只放标题，无正文。目的是先确认逻辑节奏，再填内容。

```python
# make_ppt_skeleton.py — 生成骨架 PPT，每张幻灯片只有标题
from pptx import Presentation
from pptx.util import Inches, Pt
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN

SLIDES = [
    # (标题, 副标题/备注)
    ("vLLM vs SGLang：推理框架深度对比",
     "实测数据 · RTX 3060 · Qwen2.5-1.5B\n[你的名字] · 2026-07"),
    ("目录",
     "① 为什么需要推理框架\n② vLLM：三层优化\n③ SGLang：缓存粒度革新\n④ 对比数据\n⑤ 选型建议"),
    # Part 1
    ("GPU 利用率：朴素推理在浪费什么？", "decode 阶段 GPU 利用率 <40%，原因是..."),
    ("KV Cache：推理的核心瓶颈", "每个 token 都要看所有历史 KV，内存读写是瓶颈"),
    ("传统 KV Cache 的内存浪费", "预分配 → 碎片 + 浪费，vLLM 之前业界如何处理"),
    # Part 2
    ("PagedAttention：向操作系统学内存管理", "Block Table + 16-token KV block，浪费从60%→<4%"),
    ("Continuous Batching：动态插入，消灭气泡", "iteration-level scheduling，旧序列完成立刻补新序列"),
    ("APC：前缀只算一次，多次复用", "chain hash + LRU eviction，实测 TTFT 降幅"),
    ("vLLM 数据汇总", "三层优化对应三张数据图"),
    # Part 3
    ("RadixAttention：缓存粒度降到 token 级", "Radix Tree vs Block Table：命中粒度差异"),
    ("Compressed FSM：结构化输出的跳步加速", "状态机压缩 + jump-forward，避免逐 token mask"),
    ("HiCache：GPU→CPU→Disk 三级缓存", "面向超长上下文和高复用场景"),
    ("SGLang 实验数据", "多轮对话 TTFT，前缀复用率实验"),
    # Part 4
    ("通用负载：两框架基本持平", "场景A fixed-output，并发1-20"),
    ("高复用负载：SGLang 显著更优", "多轮对话轮次5，TTFT 差异 X%"),
    ("核心差异：缓存命中粒度", "kvcache_diff.png"),
    ("前缀复用率实验（0→100%）", "prefix_ratio_comparison.png"),
    # Part 5
    ("选型决策树", "selection_guide.png"),
    ("一句话结论", "高复用→SGLang，通用→vLLM，差异在缓存粒度"),
    ("Q&A", "感谢聆听"),
]

def make_skeleton(output_path="学习日志/PPT骨架.pptx"):
    prs = Presentation()
    prs.slide_width  = Inches(13.33)
    prs.slide_height = Inches(7.5)

    blank_layout = prs.slide_layouts[6]  # 全空白

    for i, (title, note) in enumerate(SLIDES):
        slide = prs.slides.add_slide(blank_layout)

        # 幻灯片编号
        txb_num = slide.shapes.add_textbox(Inches(0.2), Inches(0.1), Inches(0.8), Inches(0.4))
        txb_num.text_frame.text = str(i + 1)
        txb_num.text_frame.paragraphs[0].runs[0].font.size = Pt(12)
        txb_num.text_frame.paragraphs[0].runs[0].font.color.rgb = RGBColor(0xAA,0xAA,0xAA)

        # 标题
        txb = slide.shapes.add_textbox(Inches(1), Inches(2.5), Inches(11), Inches(1.5))
        tf = txb.text_frame
        tf.word_wrap = True
        p = tf.paragraphs[0]
        p.text = title
        p.alignment = PP_ALIGN.CENTER
        p.runs[0].font.size = Pt(36)
        p.runs[0].font.bold = True
        p.runs[0].font.color.rgb = RGBColor(0x1F,0x4E,0x79)

        # 备注（灰色小字）
        if note:
            txb2 = slide.shapes.add_textbox(Inches(1.5), Inches(4.2), Inches(10), Inches(2))
            tf2 = txb2.text_frame
            tf2.word_wrap = True
            p2 = tf2.paragraphs[0]
            p2.text = note
            p2.alignment = PP_ALIGN.CENTER
            p2.runs[0].font.size = Pt(18)
            p2.runs[0].font.color.rgb = RGBColor(0x70,0x70,0x70)

        # 幻灯片备注（演讲者视图）
        slide.notes_slide.notes_text_frame.text = note

    prs.save(output_path)
    print(f"骨架 PPT 已保存：{output_path}（{len(SLIDES)} 张幻灯片）")

if __name__ == "__main__":
    make_skeleton()
```
