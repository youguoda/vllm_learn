## 步骤 1：按场景重组数据（40 min）

```python
# summarize_by_scenario.py — 把所有数据按"对外讲的场景"重新组织
import json, csv
from pathlib import Path

with open("bench_results/all_data.json") as f:
    D = json.load(f)

if __name__ == "__main__":
    print("=" * 60)
    print("场景1：通用批量推理（无共享前缀）")
    print("=" * 60)
    for fw in ("vllm","sglang"):
        rows = D.get("fixed_output",{}).get(fw,[])
        if not rows: continue
        best_tps = max(r["tok_per_s"] for r in rows)
        low_conc = next(r for r in rows if int(r["concurrency"]) == 1)
        print(f"  {fw}: 最高吞吐={best_tps:.1f} tok/s, "
              f"单并发TTFT={low_conc['ttft_median']:.0f}ms")

    print("\n" + "=" * 60)
    print("场景2：多轮对话（高前缀复用）")
    print("=" * 60)
    for fw in ("vllm","sglang"):
        rows = [r for r in D.get("multi_turn",{}).get(fw,[])
                if r.get("concurrency")==8 and str(r.get("turn","")).isdigit()]
        if not rows: continue
        t1 = next((r["ttft_median"] for r in rows if int(r["turn"])==1), None)
        t3 = next((r["ttft_median"] for r in rows if int(r["turn"])==3), None)
        print(f"  {fw}: 轮1 TTFT={t1:.0f}ms, 轮3 TTFT={t3:.0f}ms" if t1 and t3 else f"  {fw}: 无数据")

    print("\n" + "=" * 60)
    print("场景3：ITL（打字体验）")
    print("=" * 60)
    for fw in ("vllm","sglang"):
        rows = D.get("streaming",{}).get(fw,[])
        row8 = next((r for r in rows if r.get("concurrency")==8), None)
        if row8:
            print(f"  {fw} 并发8: P50={row8['itl_p50']:.1f}ms, P99={row8['itl_p99']:.1f}ms")

    print("\n" + "=" * 60)
    print("关键对比数字（用于PPT结论页）")
    print("=" * 60)
    # 多轮对话 SGLang vs vLLM 轮次5的差异
    results = {}
    for fw in ("vllm","sglang"):
        rows = [r for r in D.get("multi_turn",{}).get(fw,[])
                if r.get("concurrency")==8 and str(r.get("turn","")).isdigit()]
        t5 = next((r["ttft_median"] for r in rows if int(r["turn"])==5), None)
        results[fw] = t5
    if all(v is not None for v in results.values()):
        diff = results["vllm"] - results["sglang"]
        pct  = diff / results["vllm"] * 100
        print(f"  多轮5轮后 TTFT：vLLM={results['vllm']:.0f}ms, "
              f"SGLang={results['sglang']:.0f}ms, SGLang快{pct:.0f}%")
```
