# 05-04 详细计划：vLLM vs SGLang 初步跑分

## 步骤 1：框架无关压测脚本（30 min）

新建 `08_compare_bench.py`（基于 04_continuous_batching.py 改造，只依赖 OpenAI 兼容接口）：

```python
"""
vLLM vs SGLang 对比压测（OpenAI 接口通用）
用法:
  NO_PROXY="*" python3 08_compare_bench.py --base-url http://localhost:8000/v1  --tag vllm
  NO_PROXY="*" python3 08_compare_bench.py --base-url http://localhost:30000/v1 --tag sglang
"""
import argparse, asyncio, csv, time
import openai

PROMPTS = [f"请用约100字介绍主题{i}：工业传感器的一种典型应用。" for i in range(40)]

async def one(client, model, prompt):
    t0 = time.time()
    ttft = None; ntok = 0
    r = await client.chat.completions.create(model=model,
        messages=[{"role": "user", "content": prompt}], max_tokens=128, stream=True)
    async for c in r:
        if ttft is None: ttft = time.time() - t0
        if c.choices and c.choices[0].delta.content: ntok += 1
    e2e = time.time() - t0
    itl = (e2e - ttft) / max(ntok - 1, 1)
    return ttft, e2e, itl, ntok

async def bench(base, model, conc):
    client = openai.AsyncOpenAI(base_url=base, api_key="x")
    sem = asyncio.Semaphore(conc)
    async def guarded(p):
        async with sem: return await one(client, model, p)
    t0 = time.time()
    rs = await asyncio.gather(*[guarded(p) for p in PROMPTS])
    wall = time.time() - t0
    n = len(rs)
    return {"concurrency": conc, "rps": n / wall,
            "ttft_avg": sum(r[0] for r in rs) / n,
            "itl_avg": sum(r[2] for r in rs) / n,
            "tok_per_s": sum(r[3] for r in rs) / wall}

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--base-url", required=True)
    ap.add_argument("--model", default="Qwen/Qwen2.5-1.5B")
    ap.add_argument("--tag", required=True)
    a = ap.parse_args()
    rows = []
    asyncio.run(bench(a.base_url, a.model, 2))  # 预热，不计
    for c in (1, 5, 10, 20):
        r = asyncio.run(bench(a.base_url, a.model, c))
        r["framework"] = a.tag
        print(r); rows.append(r)
    with open(f"bench_{a.tag}.csv", "w", newline="") as f:
        w = csv.DictWriter(f, fieldnames=rows[0].keys()); w.writeheader(); w.writerows(rows)
```

## 步骤 3：画图（30 min）

```python
# plot_compare.py
import csv, matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

def load(tag):
    with open(f"bench_{tag}.csv") as f:
        return list(csv.DictReader(f))

if __name__ == "__main__":
    fig, (a1, a2) = plt.subplots(1, 2, figsize=(11, 4))
    for tag, style in (("vllm", "o-"), ("sglang", "s--")):
        rows = load(tag)
        x = [int(r["concurrency"]) for r in rows]
        a1.plot(x, [float(r["tok_per_s"]) for r in rows], style, label=tag)
        a2.plot(x, [float(r["ttft_avg"])*1000 for r in rows], style, label=tag)
    a1.set_title("吞吐 (token/s)"); a2.set_title("TTFT (ms)")
    for a in (a1, a2): a.set_xlabel("并发"); a.legend()
    fig.savefig("assets/compare_v1.png", dpi=150, bbox_inches="tight")
```
