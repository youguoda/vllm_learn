# 05-10 详细计划：前缀复用率 vs 性能实验（全项目最重要数据）

## 步骤 1：实验脚本（60 min）

新建 `09_prefix_ratio_bench.py`。核心思路：控制"共享前缀占总输入的比例"，其余条件完全相同，对比 vLLM 和 SGLang 的 TTFT 和吞吐。

```python
"""
前缀复用率实验
变量：prefix_ratio = 共享前缀长度 / 总输入长度
固定：总输入 ~512 token，输出 64 token，20 个请求，并发 8

启动 vLLM:  HF_HUB_OFFLINE=1 NO_PROXY="*" vllm serve Qwen/Qwen2.5-1.5B \
              --max-model-len 4096 --gpu-memory-utilization 0.85
启动 SGLang: NO_PROXY="*" HF_HUB_OFFLINE=1 python3 -m sglang.launch_server \
              --model-path Qwen/Qwen2.5-1.5B --context-length 4096 \
              --mem-fraction-static 0.85 --port 30000
"""
import asyncio, csv, time, argparse
import openai

# 构造 system prompt 内容（固定，长度可控）
BASE_SYS = ("你是工业质检专家，精通以下知识：振动分析、频谱特征、故障模式识别、"
            "统计过程控制、六西格玛方法论、预测性维护策略。") * 5  # ~250 token

UNIQUE_QUESTIONS = [
    f"请分析场景{i}：传感器编号S{i:03d}的振动超标，频率为{300+i*7}Hz，给出诊断建议。"
    for i in range(20)
]

def build_messages(prefix_ratio: float, question: str):
    """
    prefix_ratio=1.0: 所有请求用同一 system prompt（100% 可复用）
    prefix_ratio=0.0: system prompt 包含 question 内容（0% 共享）
    prefix_ratio=0.5: system prompt 固定部分 + 问题相关部分各占一半
    """
    shared_len = int(len(BASE_SYS) * prefix_ratio)
    shared = BASE_SYS[:shared_len]
    unique_sys = question[:len(BASE_SYS) - shared_len]  # 剩余用问题内容填
    system = shared + unique_sys
    return [{"role": "system", "content": system},
            {"role": "user", "content": question}]

async def one(client, model, msgs):
    t0 = time.time()
    ttft = None; ntok = 0
    r = await client.chat.completions.create(
        model=model, messages=msgs, max_tokens=64, stream=True)
    async for c in r:
        if ttft is None: ttft = time.time() - t0
        if c.choices and c.choices[0].delta.content: ntok += 1
    return ttft, time.time() - t0, ntok

async def run_ratio(base_url, model, ratio, n_concur=8):
    client = openai.AsyncOpenAI(base_url=base_url, api_key="x")
    sem = asyncio.Semaphore(n_concur)
    all_msgs = [build_messages(ratio, q) for q in UNIQUE_QUESTIONS]
    async def guarded(msgs):
        async with sem: return await one(client, model, msgs)
    t0 = time.time()
    rs = await asyncio.gather(*[guarded(m) for m in all_msgs])
    wall = time.time() - t0
    n = len(rs)
    return {
        "prefix_ratio": ratio,
        "ttft_avg_ms": sum(r[0] for r in rs) / n * 1000,
        "ttft_p95_ms": sorted(r[0] for r in rs)[int(n*0.95)] * 1000,
        "rps": n / wall,
    }

if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--base-url", required=True)
    ap.add_argument("--model", default="Qwen/Qwen2.5-1.5B")
    ap.add_argument("--tag", required=True)
    a = ap.parse_args()

    rows = []
    # 预热
    asyncio.run(run_ratio(a.base_url, a.model, 1.0, n_concur=4))
    for ratio in (0.0, 0.25, 0.5, 0.75, 1.0):
        r = asyncio.run(run_ratio(a.base_url, a.model, ratio))
        r["framework"] = a.tag
        print(f"ratio={ratio:.2f}  TTFT={r['ttft_avg_ms']:.1f}ms  RPS={r['rps']:.2f}")
        rows.append(r)

    with open(f"prefix_ratio_{a.tag}.csv", "w", newline="") as f:
        w = csv.DictWriter(f, fieldnames=rows[0].keys())
        w.writeheader(); w.writerows(rows)
```

## 步骤 3：画图（30 min）

```python
# plot_prefix_ratio.py
import csv, matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

def load(tag):
    with open(f"prefix_ratio_{tag}.csv") as f:
        return list(csv.DictReader(f))

if __name__ == "__main__":
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
    colors = {"vllm": ("#4C72B0", "o-"), "sglang": ("#55A868", "s--")}
    for tag, (c, sty) in colors.items():
        rows = load(tag)
        x = [float(r["prefix_ratio"]) for r in rows]
        ax1.plot(x, [float(r["ttft_avg_ms"]) for r in rows], sty, color=c, label=tag, ms=7)
        ax2.plot(x, [float(r["rps"]) for r in rows], sty, color=c, label=tag, ms=7)

    ax1.set_title("前缀复用率 vs 平均 TTFT")
    ax1.set_xlabel("前缀复用率"); ax1.set_ylabel("TTFT (ms)")
    ax2.set_title("前缀复用率 vs 吞吐 (RPS)")
    ax2.set_xlabel("前缀复用率"); ax2.set_ylabel("RPS")
    for a in (ax1, ax2): a.legend(); a.set_xticks([0,.25,.5,.75,1.0])
    ax1.invert_xaxis()  # 左高右低更直观

    fig.suptitle("vLLM vs SGLang：前缀复用率对性能的影响", fontsize=13)
    fig.tight_layout()
    fig.savefig("assets/prefix_ratio_comparison.png", dpi=150, bbox_inches="tight")
```

**预期形状**：

```
TTFT 图：
  ratio=0%  两框架 TTFT 相近（都要算全部）
  ratio↑   两者都下降，SGLang 下降更陡（粒度更细，命中更多）
  ratio=100% SGLang TTFT 最低，vLLM 也低但略高（有16-token 对齐损耗）

吞吐图：
  ratio↑   吞吐都上升（省了 prefill，GPU 时间更多给 decode）
```
