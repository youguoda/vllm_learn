# draw_pagedattention.py
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
from matplotlib.patches import FancyArrowPatch
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']
plt.rcParams['axes.unicode_minus'] = False

if __name__ == "__main__":
    fig, ax = plt.subplots(figsize=(13, 7))
    ax.set_xlim(0, 13); ax.set_ylim(0, 7); ax.axis('off')
    ax.set_title("vLLM PagedAttention：KV Cache 管理全流程", fontsize=14, pad=12)

    def box(x, y, w, h, txt, color):
        ax.add_patch(mpatches.FancyBboxPatch((x,y), w, h,
            boxstyle="round,pad=0.1", fc=color, ec="gray", lw=1.2))
        ax.text(x+w/2, y+h/2, txt, ha='center', va='center', fontsize=9)

    def arrow(x1,y1,x2,y2,lbl=""):
        ax.annotate("", xy=(x2,y2), xytext=(x1,y1),
            arrowprops=dict(arrowstyle="->", color="steelblue", lw=1.5))
        if lbl: ax.text((x1+x2)/2, (y1+y2)/2+0.15, lbl, fontsize=7.5, color="steelblue", ha='center')

    # 请求入队
    box(0.3, 5.2, 2.2, 0.8, "新请求\n(token ids)", "#dde8f5")
    # allocate_slots
    box(3.0, 5.2, 2.5, 0.8, "allocate_slots()\n查 APC 哈希表", "#fff3cd")
    arrow(2.5, 5.6, 3.0, 5.6, "")
    # 命中分支
    box(3.0, 3.8, 2.5, 0.8, "命中 → 复用物理块\n(跳过 prefill)", "#d4edda")
    ax.text(4.25, 4.95, "命中", fontsize=8, color="green", ha='center')
    arrow(4.25, 5.2, 4.25, 4.6, "")
    # 未命中分支
    box(6.2, 3.8, 2.5, 0.8, "未命中 → free_list\n取新物理块", "#f8d7da")
    ax.text(7.45, 4.95, "未命中", fontsize=8, color="red", ha='center')
    arrow(5.5, 5.6, 6.2, 4.2, "")
    # Block Table
    box(5.0, 2.2, 2.8, 0.9, "Block Table\n[逻辑块 → 物理块 id]", "#e2d9f3")
    arrow(4.25, 3.8, 5.5, 3.1, "")
    arrow(7.45, 3.8, 6.8, 3.1, "")
    # Prefill/Decode
    box(8.5, 3.8, 2.5, 0.8, "Prefill + Decode\n(attention kernel)", "#cfe2ff")
    arrow(7.8, 2.65, 8.5, 4.0, "Block Table 传入")
    # 满块 → 计算哈希
    box(8.5, 2.2, 2.5, 0.8, "块满 → _compute_hash()\n写入 block.block_hash", "#fff3cd")
    arrow(9.75, 3.8, 9.75, 3.0, "")
    # EOS → 释放
    box(8.5, 0.7, 2.5, 0.8, "EOS → free_blocks()\nref_count-1 → free_list\n(哈希保留供复用)", "#d4edda")
    arrow(9.75, 2.2, 9.75, 1.5, "")
    # 物理块池
    box(0.3, 2.2, 2.2, 1.5, "GPU 物理块池\n(free_list)\nnum_gpu_blocks 个", "#f5f5f5")
    arrow(0.3+2.2, 2.95, 6.2, 4.2, "取块")
    arrow(8.5, 1.1, 2.5, 2.95, "归还")

    # 图注
    ax.text(0.3, 0.3,
        "关键性质：①逻辑连续/物理离散  ②按需分配(浪费<4%)  ③链式哈希保证APC正确性",
        fontsize=8.5, style='italic', color='dimgray')

    fig.tight_layout()
    fig.savefig("assets/pagedattention_arch.png", dpi=150, bbox_inches="tight")
    print("已保存 assets/pagedattention_arch.png")
```

# draw_diff.py
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, axes = plt.subplots(1, 3, figsize=(14, 4.5))
    fig.suptitle("PagedAttention vs RadixAttention：三维差异", fontsize=13)

    # 子图1：缓存粒度
    ax = axes[0]; ax.set_title("①命中粒度")
    ax.barh(["vLLM\n(块哈希)", "SGLang\n(radix树)"], [16, 1],
            color=["#f8d7da", "#d4edda"])
    ax.set_xlabel("最小命中单位 (token)"); ax.axvline(1, color='gray', ls='--', lw=0.8)

    # 子图2：共享前缀场景 TTFT（填入 05-02 数据）
    ax = axes[1]; ax.set_title("②共享前缀 TTFT\n(1000 token 前缀, 第2条起)")
    ttft_data = {
        "vLLM\nAPC": 33.3,       # 04-26 实验数据
        "SGLang\nRadix": 0,      # 填 05-02 数据
    }
    ax.bar(ttft_data.keys(), ttft_data.values(), color=["#dde8f5", "#d4edda"])
    ax.set_ylabel("TTFT (ms)"); ax.set_ylim(0, 200)
    ax.text(0, 33.3+5, "33ms", ha='center', fontsize=9)

    # 子图3：半块前缀（05-02/04-29 数据）
    ax = axes[2]; ax.set_title("③半块对齐前缀\n(前缀长度=18 token, 块大小16)")
    ax.bar(["vLLM\n(命中16)", "SGLang\n(命中18)"], [16, 18], color=["#f8d7da", "#d4edda"])
    ax.set_ylabel("实际命中 token 数")
    ax.text(0, 16.3, "仅16", ha='center', fontsize=9, color='red')
    ax.text(1, 18.3, "全18", ha='center', fontsize=9, color='green')

    fig.tight_layout()
    fig.savefig("assets/kvcache_diff.png", dpi=150, bbox_inches="tight")
    print("已保存 assets/kvcache_diff.png")
```
