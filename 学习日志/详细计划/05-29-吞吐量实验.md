```bash
# run_fixed_output_3x.sh
#!/bin/bash
FRAMEWORK=$1  # vllm 或 sglang
PORT=${2:-8000}

for round in 1 2 3; do
    echo "=== Round $round ==="
    NO_PROXY="*" python3 12_bench_runner.py \
      --framework $FRAMEWORK --scenario fixed_output \
      --output-suffix "_r${round}"
    echo "Round $round 完成，休息 10s..."
    sleep 10
done
echo "全部完成，合并结果..."
python3 merge_results.py $FRAMEWORK fixed_output
```

先在 `12_bench_runner.py` 加 `--output-suffix` 参数，让三轮结果分别存 `_r1/_r2/_r3`。

**合并脚本**：

```python
# merge_results.py — 三轮取中位数
import csv, statistics, sys
from pathlib import Path

def merge(framework, scenario):
    OUTPUT_DIR = Path("bench_results")
    # 读三轮
    all_rounds = []
    for r in (1, 2, 3):
        p = OUTPUT_DIR / f"{framework}_{scenario}_r{r}.csv"
        if not p.exists():
            print(f"缺少 {p}，跳过"); continue
        with open(p) as f: all_rounds.append(list(csv.DictReader(f)))

    if not all_rounds: return

    # 按并发对齐，取中位数
    keys = list(all_rounds[0][0].keys())
    merged = []
    for i, row0 in enumerate(all_rounds[0]):
        m = {"framework": framework, "scenario": scenario}
        m["concurrency"] = row0["concurrency"]
        for k in keys:
            if k in ("concurrency", "framework"): continue
            try:
                vals = [float(rd[i][k]) for rd in all_rounds if i < len(rd)]
                m[k] = statistics.median(vals)
            except (ValueError, KeyError):
                m[k] = row0.get(k, "")
        merged.append(m)

    out = OUTPUT_DIR / f"{framework}_{scenario}.csv"
    with open(out, "w", newline="") as f:
        w = csv.DictWriter(f, fieldnames=merged[0].keys())
        w.writeheader(); w.writerows(merged)
    print(f"合并完成 → {out}")

if __name__ == "__main__":
    merge(sys.argv[1], sys.argv[2])
```

## 步骤 2：找吞吐拐点（30 min）

跑完立刻画初版吞吐曲线：

```python
# quick_plot_throughput.py
import csv, matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

def load(fw):
    with open(f"bench_results/{fw}_fixed_output.csv") as f:
        return list(csv.DictReader(f))

if __name__ == "__main__":
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))
    for fw, c, s in (("vllm","#4C72B0","o-"), ("sglang","#55A868","s--")):
        try:
            rows = load(fw)
            x  = [int(r["concurrency"]) for r in rows]
            y1 = [float(r["tok_per_s"])    for r in rows]
            y2 = [float(r["ttft_median"])  for r in rows]
            ax1.plot(x, y1, s, color=c, label=fw, ms=7)
            ax2.plot(x, y2, s, color=c, label=fw, ms=7)
            # 找吞吐拐点（吞吐增速 <5% 的第一个点）
            for i in range(1, len(y1)):
                growth = (y1[i] - y1[i-1]) / y1[i-1]
                if growth < 0.05:
                    ax1.axvline(x[i], color=c, ls=':', lw=1)
                    ax1.text(x[i]+0.3, y1[i], f'拐点\n(并发={x[i]})',
                             fontsize=7, color=c)
                    break
        except FileNotFoundError:
            print(f"{fw} 数据不存在，跳过")

    ax1.set_title("吞吐量 (token/s)"); ax1.set_xlabel("并发数"); ax1.legend()
    ax2.set_title("TTFT 中位数 (ms)"); ax2.set_xlabel("并发数"); ax2.legend()
    fig.suptitle("vLLM vs SGLang 吞吐/延迟（固定输出场景）", fontsize=13)
    fig.tight_layout()
    fig.savefig("assets/throughput_v1.png", dpi=150, bbox_inches="tight")
    print("保存 assets/throughput_v1.png")
```

## 步骤 3：拐点分析（20 min）

在笔记里写清楚拐点的物理含义：

- **拐点之前**：新加的并发请求能被 GPU 真正"同时处理"，吞吐近线性增长
- **拐点之后**：GPU 已经饱和（或 KV blocks 不够），新请求排队，吞吐增长趋缓，TTFT 开始飙升
- **对于 12GB 单卡 + 1.5B 模型**：拐点大约在并发 8-16（取决于实际 KV block 数）

如果两框架拐点不同（如 SGLang 在更高并发才拐），说明 SGLang 的调度或内存使用更高效。
