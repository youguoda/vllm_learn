# draw_scheduler_compare.py
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 7))
    fig.suptitle("调度策略对比：vLLM vs SGLang", fontsize=14)

    def box(ax, x, y, w, h, txt, color, fontsize=9):
        ax.add_patch(mpatches.FancyBboxPatch((x,y), w, h,
            boxstyle="round,pad=0.1", fc=color, ec="gray", lw=1.2))
        ax.text(x+w/2, y+h/2, txt, ha='center', va='center', fontsize=fontsize)

    def arr(ax, x1, y1, x2, y2, lbl="", color="steelblue"):
        ax.annotate("", xy=(x2,y2), xytext=(x1,y1),
            arrowprops=dict(arrowstyle="->", color=color, lw=1.4))
        if lbl:
            ax.text((x1+x2)/2+0.1, (y1+y2)/2, lbl, fontsize=8,
                    color=color, ha='center')

    for ax, title in [(ax1, "vLLM 调度器"), (ax2, "SGLang 调度器")]:
        ax.set_xlim(0,8); ax.set_ylim(0,8); ax.axis('off')
        ax.set_title(title, fontsize=12)

    # vLLM
    box(ax1, 2.5,6.8,3.0,0.7,"新请求","#dde8f5")
    box(ax1, 2.5,5.6,3.0,0.7,"WAITING 队列\n(FCFS)","#fff3cd")
    arr(ax1, 4.0,6.8,4.0,6.3)
    box(ax1, 2.5,4.3,3.0,0.7,"Token Budget 检查\n(max_num_batched_tokens)","#ffe0b2")
    arr(ax1, 4.0,5.6,4.0,5.0)
    box(ax1, 2.5,3.0,3.0,0.7,"RUNNING 队列\n(Continuous Batching)","#d4edda")
    arr(ax1, 4.0,4.3,4.0,3.7)
    box(ax1, 0.3,1.5,2.8,0.9,"PREEMPTED\n(recompute/swap)\n→ 回 WAITING","#f8d7da")
    box(ax1, 4.9,1.5,2.8,0.7,"FINISHED","#e2d9f3")
    arr(ax1, 4.0,3.0,2.0,2.4,"KV 不足")
    arr(ax1, 4.0,3.0,6.3,2.2,"EOS")
    ax1.text(0.2,0.5,"特点：FCFS调度，APC查表，抢占恢复靠重算/swap",
             fontsize=8, style='italic', color='dimgray')

    # SGLang
    box(ax2, 2.5,6.8,3.0,0.7,"新请求","#dde8f5")
    box(ax2, 2.5,5.6,3.0,0.7,"Radix 树查询\n(match_prefix)","#fff3cd")
    arr(ax2, 4.0,6.8,4.0,6.3)
    box(ax2, 0.3,4.3,2.8,0.9,"命中 → 优先调度\n(cache-aware scheduling)\n减少 prefill","#d4edda")
    box(ax2, 4.9,4.3,2.8,0.9,"未命中 → 普通排队\n分配 KV Pool 内存","#fff3cd")
    arr(ax2, 4.0,5.6,1.7,5.2,"命中")
    arr(ax2, 4.0,5.6,6.3,5.2,"未命中")
    box(ax2, 2.5,3.0,3.0,0.7,"RUNNING\n(Continuous Batching)","#d4edda")
    arr(ax2, 1.7,4.3,3.5,3.7)
    arr(ax2, 6.3,4.3,4.5,3.7)
    box(ax2, 2.5,1.5,3.0,0.7,"EOS → insert 写回 radix 树\n→ FINISHED","#e2d9f3")
    arr(ax2, 4.0,3.0,4.0,2.2)
    ax2.text(0.2,0.5,"特点：缓存感知调度（优先命中缓存的请求），淘汰靠 evict",
             fontsize=8, style='italic', color='dimgray')

    fig.tight_layout()
    fig.savefig("assets/scheduler_compare.png", dpi=150, bbox_inches="tight")
    print("已保存 assets/scheduler_compare.png")
```

**关键差异标注**（写进图注或配套笔记）：vLLM 的调度是"公平 FCFS，遇到资源不足才抢占"；SGLang 是"缓存感知调度，优先调度能命中 radix 树的请求"——后者在高复用负载下能减少 prefill 量，但实现更复杂。

# draw_parallelism_table.py
import matplotlib.pyplot as plt
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']

if __name__ == "__main__":
    fig, ax = plt.subplots(figsize=(14, 5))
    ax.axis('off')

    headers = ["并行方式", "切分对象", "通信原语", "通信量(相对)", "显存节省", "典型场景"]
    rows = [
        ["TP\nTensor Parallel",  "单层权重\n(行/列切)",   "All-Reduce",  "高\n(每层1次)",    "权重 ÷N",    "单机多卡\n7B~70B"],
        ["PP\nPipeline Parallel","层\n(按深度切)",        "P2P send/recv","低\n(只传激活)",  "权重 ÷p",    "跨机\n超大模型"],
        ["EP\nExpert Parallel",  "MoE 专家\n(按编号)",    "All-to-All",  "中\n(定向路由)",   "专家权重 ÷N","MoE 模型\nDeepSeek/Mixtral"],
        ["DP\nData Parallel",    "请求/数据\n(按样本)",   "外部 LB",     "最低\n(无KV共享)", "无",         "推理扩容\n多实例"],
    ]
    colors_row = ["#dde8f5","#d4edda","#fff3cd","#fde8d8"]

    table = ax.table(
        cellText=rows,
        colLabels=headers,
        cellLoc='center', loc='center',
        bbox=[0, 0, 1, 1]
    )
    table.auto_set_font_size(False)
    table.set_fontsize(10)
    for (r, c), cell in table.get_celld().items():
        if r == 0:
            cell.set_facecolor('#2c3e50'); cell.set_text_props(color='white', fontsize=10)
        elif r > 0:
            cell.set_facecolor(colors_row[r-1])
        cell.set_edgecolor('white')
        cell.set_height(0.18)

    ax.set_title("推理框架并行策略速查表（vLLM & SGLang）", fontsize=13, pad=15)
    fig.tight_layout()
    fig.savefig("assets/parallelism_table.png", dpi=150, bbox_inches="tight")
```

## 步骤 3：决策图（30 min）

存 `assets/parallelism_decision.png`，一张"什么情况用什么并行"的流程图：

关键分支逻辑：
- 单卡放得下？→ 不用并行
- 单机多卡，Dense 模型？→ TP
- 跨机？→ TP × PP 组合
- MoE 模型？→ EP + TP（Attention 用 TP，Expert 用 EP）
- 推理扩容（不是单请求延迟）？→ DP（多实例）

用 draw.io 或 matplotlib 均可，PPT 里只需要截图，清晰即可。
