## 步骤 1：数据合并清洗（40 min）

```python
# prepare_analysis.py — 合并所有 CSV，统一格式，输出一个主数据集
import csv, json
from pathlib import Path

SCENARIOS = {
    "fixed_output": ["concurrency","rps","tok_per_s","ttft_median","ttft_p95","itl_median","e2e_median"],
    "latency":      ["input_len_approx","ttft_median","ttft_p95","e2e_median"],
    "multi_turn":   ["concurrency","turn","ttft_median"],
    "streaming":    ["concurrency","itl_p50","itl_p95","itl_p99"],
}

if __name__ == "__main__":
    all_data = {}
    for scenario, fields in SCENARIOS.items():
        all_data[scenario] = {}
        for fw in ("vllm","sglang"):
            p = Path(f"bench_results/{fw}_{scenario}.csv")
            if not p.exists(): print(f"跳过 {p}"); continue
            with open(p) as f: rows = list(csv.DictReader(f))
            # 类型转换
            cleaned = []
            for r in rows:
                cr = {"framework": fw}
                for k in fields:
                    if k in r:
                        try: cr[k] = float(r[k])
                        except ValueError: cr[k] = r[k]
                cleaned.append(cr)
            all_data[scenario][fw] = cleaned
            print(f"  {fw} {scenario}: {len(cleaned)} 行 OK")

    with open("bench_results/all_data.json", "w") as f:
        json.dump(all_data, f, ensure_ascii=False, indent=2)
    print("\nall_data.json 已保存")
```

## 步骤 2：四张主图（60 min）

新建 `13_plot_all.py`，一键生成全部分析图：

```python
"""
13_plot_all.py — 生成全部对比图表
运行: python3 13_plot_all.py
输出: assets/ 下 8 张图
"""
import json, matplotlib.pyplot as plt
import matplotlib.ticker as mticker
from pathlib import Path
plt.rcParams['font.sans-serif'] = ['Microsoft YaHei', 'SimHei']
plt.rcParams['axes.unicode_minus'] = False
Path("assets").mkdir(exist_ok=True)

with open("bench_results/all_data.json") as f:
    D = json.load(f)

COLORS = {"vllm": "#4C72B0", "sglang": "#55A868"}
STYLES = {"vllm": "o-",      "sglang": "s--"}

def get(scenario, fw, key):
    rows = D.get(scenario,{}).get(fw,[])
    return [r[key] for r in rows if key in r]

def getx(scenario, fw, xkey, ykey, filter_fn=None):
    rows = D.get(scenario,{}).get(fw,[])
    if filter_fn: rows = [r for r in rows if filter_fn(r)]
    return [r[xkey] for r in rows if xkey in r and ykey in r], \
           [r[ykey] for r in rows if xkey in r and ykey in r]

# ── 图1：吞吐量 ──────────────────────────────────────────────
fig, ax = plt.subplots(figsize=(8,5))
for fw in ("vllm","sglang"):
    x, y = getx("fixed_output", fw, "concurrency", "tok_per_s")
    if x: ax.plot(x, y, STYLES[fw], color=COLORS[fw], label=fw, ms=8, lw=2)
ax.set_xlabel("并发请求数"); ax.set_ylabel("吞吐量 (token/s)")
ax.set_title("吞吐量对比（固定输出 128 token）")
ax.legend(); ax.grid(alpha=0.3)
fig.tight_layout(); fig.savefig("assets/p1_throughput.png", dpi=150, bbox_inches="tight")
plt.close()

# ── 图2：并发-TTFT ────────────────────────────────────────────
fig, ax = plt.subplots(figsize=(8,5))
for fw in ("vllm","sglang"):
    x, y = getx("fixed_output", fw, "concurrency", "ttft_median")
    if x: ax.plot(x, y, STYLES[fw], color=COLORS[fw], label=fw, ms=8, lw=2)
ax.set_xlabel("并发请求数"); ax.set_ylabel("TTFT 中位数 (ms)")
ax.set_title("首 Token 延迟 vs 并发数")
ax.legend(); ax.grid(alpha=0.3)
fig.tight_layout(); fig.savefig("assets/p2_ttft_vs_conc.png", dpi=150, bbox_inches="tight")
plt.close()

# ── 图3：TTFT vs 输入长度 ─────────────────────────────────────
fig, ax = plt.subplots(figsize=(8,5))
for fw in ("vllm","sglang"):
    x, y = getx("latency", fw, "input_len_approx", "ttft_median")
    if x: ax.plot(x, y, STYLES[fw], color=COLORS[fw], label=fw, ms=8, lw=2)
ax.set_xlabel("输入长度（近似 token 数）"); ax.set_ylabel("TTFT 中位数 (ms)")
ax.set_title("首 Token 延迟 vs 输入长度（低并发，prefill 主导）")
ax.legend(); ax.grid(alpha=0.3)
fig.tight_layout(); fig.savefig("assets/p3_ttft_vs_input.png", dpi=150, bbox_inches="tight")
plt.close()

# ── 图4：多轮对话 TTFT（并发=8）─────────────────────────────
fig, ax = plt.subplots(figsize=(8,5))
for fw in ("vllm","sglang"):
    rows = [r for r in D.get("multi_turn",{}).get(fw,[])
            if r.get("concurrency")==8 and str(r.get("turn","")).isdigit()]
    if rows:
        x = [int(r["turn"]) for r in rows]
        y = [r["ttft_median"] for r in rows]
        ax.plot(x, y, STYLES[fw], color=COLORS[fw], label=fw, ms=8, lw=2)
ax.set_xlabel("对话轮次"); ax.set_ylabel("TTFT 中位数 (ms)")
ax.set_title("多轮对话 TTFT（并发=8，共享 500 token system prompt）")
ax.legend(); ax.grid(alpha=0.3)
ax.annotate("SGLang RadixAttention\n从第2轮起命中缓存", xy=(2,0), fontsize=8,
            color=COLORS["sglang"], xytext=(3, 50),
            arrowprops=dict(arrowstyle="->", color=COLORS["sglang"]))
fig.tight_layout(); fig.savefig("assets/p4_multi_turn.png", dpi=150, bbox_inches="tight")
plt.close()

print("四张主图已生成：assets/p1-p4_*.png")
```

## 步骤 3：标注异常点（20 min）

检查四张图，对异常数据点各写一句解释：
- TTFT 在某并发突然跳升 → 调度器开始抢占，请求排队
- 吞吐量在高并发不涨 → GPU 饱和或 KV blocks 耗尽
- 多轮对话某轮 TTFT 异常高 → 该轮对话历史超出块边界，触发额外 prefill

这些注释写进 `学习日志/数据分析笔记.md`，不需要画进图（文字解释即可）。
