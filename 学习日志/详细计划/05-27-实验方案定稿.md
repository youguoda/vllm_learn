```python
# ── 场景 B：延迟测试（TTFT vs 输入长度）────────────────────────
def make_prompt_of_len(n_char):
    base = "请分析以下工业故障描述并给出诊断报告：" + \
           "传感器编号S042，检测到以下异常数据：振动幅值超标3倍，" * 20
    return base[:n_char] + "请给出诊断意见。"

async def bench_latency(client, input_len, n_requests=10):
    msgs = [{"role": "user", "content": make_prompt_of_len(input_len * 3)}]  # char≈token×3
    results = []
    for _ in range(n_requests):
        r = await timed_request(client, msgs, 64, stream=True)
        results.append(r)
    return {
        "input_len_approx": input_len,
        "ttft_median": statistics.median(r["ttft"] for r in results) * 1000,
        "ttft_p95":    sorted(r["ttft"] for r in results)[int(n_requests*0.95)] * 1000,
        "e2e_median":  statistics.median(r["e2e"] for r in results) * 1000,
    }

# ── 场景 C：多轮对话 ──────────────────────────────────────────
SYS_PROMPT = ("你是工业质检专家，精通振动分析、频谱分析、缺陷模式识别、"
              "六西格玛和统计过程控制。") * 8  # ~500 token

TURN_QUESTIONS = [
    "轴承外圈剥落的振动特征是什么？",
    "如何用频谱分析定位故障位置？",
    "采样频率应该如何选择？",
    "报警阈值设定有什么原则？",
    "故障发展到什么程度需要立即停机？",
]

async def single_dialog(client, sem):
    async with sem:
        msgs = [{"role": "system", "content": SYS_PROMPT}]
        ttfts = []
        for q in TURN_QUESTIONS:
            msgs.append({"role": "user", "content": q})
            r = await timed_request(client, msgs, 100, stream=True)
            ttfts.append(r["ttft"])
            msgs.append({"role": "assistant", "content": r["text"]})
        return ttfts  # 每轮的 TTFT

async def bench_multi_turn(client, concurrency, n_dialogs=20):
    sem = asyncio.Semaphore(concurrency)
    t0 = time.perf_counter()
    all_ttfts = await asyncio.gather(*[single_dialog(client, sem)
                                       for _ in range(n_dialogs)])
    wall = time.perf_counter() - t0
    # 按轮次汇总
    rows = []
    for turn_idx in range(len(TURN_QUESTIONS)):
        turn_ttfts = [d[turn_idx] for d in all_ttfts]
        rows.append({
            "concurrency": concurrency,
            "turn": turn_idx + 1,
            "ttft_median": statistics.median(turn_ttfts) * 1000,
        })
    rows.append({"concurrency": concurrency, "turn": "throughput",
                 "ttft_median": n_dialogs * len(TURN_QUESTIONS) / wall})
    return rows

# ── 场景 D：流式 ITL ──────────────────────────────────────────
async def bench_streaming(client, concurrency, n_requests=20):
    sem = asyncio.Semaphore(concurrency)
    msgs = [{"role": "user", "content": make_prompt_of_len(256*3)}]
    async def one():
        async with sem:
            return await timed_request(client, msgs, 256, stream=True)
    results = await asyncio.gather(*[one() for _ in range(n_requests)])
    itls = sorted(r["itl"] for r in results)
    return {
        "concurrency": concurrency,
        "itl_p50": itls[int(n_requests*0.50)] * 1000,
        "itl_p95": itls[int(n_requests*0.95)] * 1000,
        "itl_p99": itls[int(n_requests*0.99)] * 1000,
    }
```
